colorBar

colorfull ajax loading bar

Example

.loading()
.loaded(true)
.loaded()

More Example

Rainbow colorBar

var Rainbow = [
    ['#ff0000', '0%'], 
    ['#ffa500', '14%'], 
    ['#ffff00', '29%'], 
    ['#00ff00', '43%'], 
    ['#007fff', '57%'], 
    ['#0000ff', '72%'], 
    ['#8b00ff', '86%'], 
    ['#ff0000', '100%']
]
var bar0 = new colorBar('.bar0', {
    id: 'bar0',
    height: '5px',
    colors: Rainbow
})
.loading()
.loaded()

For Responsive Example

var bar2 = new colorBar('.bar2');

var T;
window.onresize = function() {
    clearTimeout(T)
    T = setTimeout(function() {
        if (bar2.status() == 'loading') {
            bar2.loading()
        }
    }, 500)
}
